def connect
  node = Node.find(params[:id])
  if edge = node.edges.create(target_id: params[:edge][:target_id])
    render :json => edge
  else
    render :json => { errors: edge.errors }
  end
end

def disconnect
  node = Node.find(params[:id])
  node.edges.where(target_id: params[:edge][:target_id]).destroy_all
  render :json => { status: 'success' }
end
